// Fungsi untuk menambahkan dua angka
function tambah(a, b) {
    return a + b;
}

// Fungsi untuk mengurangi dua angka
function kurang(a, b) {
    return a - b;
}

// Fungsi untuk mengalikan dua angka
function kali(a, b) {
    return a * b;
}

// Fungsi untuk membagi dua angka
function bagi(a, b) {
    if(b === 0) {
        return 'Error: Pembagi tidak boleh nol';
    } else {
        return a / b;
    }
}
